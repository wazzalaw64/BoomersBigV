<script>

  const API_URL = 'https://script.google.com/macros/s/AKfycbx416fmwllsKp-vkcgt1W7FXDRjTxu6Bzij0KewvwmVYM2YK-gTfOjon0Bo_sAQoCM2/exec';


  const videosContainer = document.getElementById('videos');

  const filterButtons = document.querySelectorAll('.filter-btn');

  const videoModal = document.getElementById('videoModal');

  const videoFrame = document.getElementById('videoFrame');

  const modalClose = document.getElementById('modalClose');


  let videosData = [];


  // Fetch data from Google Sheets API

  async function fetchVideos() {

    try {

      const response = await fetch(API_URL);

      videosData = await response.json();

      displayVideos(videosData);

    } catch (error) {

      videosContainer.innerHTML = '<p style="color:#f44336; text-align:center;">Failed to load videos.</p>';

      console.error('Error fetching videos:', error);

    }

  }


  // Display videos as cards

  function displayVideos(videos) {

    videosContainer.innerHTML = '';

    if (videos.length === 0) {

      videosContainer.innerHTML = '<p style="text-align:center;">No videos found.</p>';

      return;

    }

    videos.forEach(video => {

      const card = document.createElement('div');

      card.classList.add('card');

      card.innerHTML = `

        <img class="thumbnail" src="${video['Thumbnail URL']}" alt="${video['Game Title']}" />

        <div class="card-content">

          <h3 class="game-title">${video['Game Title']}</h3>

          <div class="game-info">${video.Status} | ${video.Team} | ${video.Date}</div>

        </div>`;

      card.addEventListener('click', () => openModal(video['YouTube Link']));

      videosContainer.appendChild(card);

    });

  }


  // Open video modal and play video

  function openModal(videoUrl) {

    const embedUrl = convertYoutubeUrlToEmbed(videoUrl);

    videoFrame.src = embedUrl + '?autoplay=1&rel=0';

    videoModal.style.display = 'flex';

  }


  // Close modal and stop video

  modalClose.addEventListener('click', () => {

    videoModal.style.display = 'none';

    videoFrame.src = '';

  });


  // Close modal on outside click

  window.addEventListener('click', (e) => {

    if (e.target === videoModal) {

      videoModal.style.display = 'none';

      videoFrame.src = '';

    }

  });


  // Convert YouTube watch URL to embed URL

  function convertYoutubeUrlToEmbed(url) {

    const videoIdMatch = url.match(/[?&]v=([^&]+)/);

    if (videoIdMatch && videoIdMatch[1]) {

      return 'https://www.youtube.com/embed/' + videoIdMatch[1];

    } else {

      // fallback to original URL if no match

      return url;

    }

  }


  // Filter videos by status

  filterButtons.forEach(btn => {

    btn.addEventListener('click', () => {

      filterButtons.forEach(b => b.classList.remove('active'));

      btn.classList.add('active');

      const filter = btn.getAttribute('data-filter');

      if (filter === 'All') {

        displayVideos(videosData);

      } else {

        const filtered = videosData.filter(video => video.Status === filter);

        displayVideos(filtered);

      }

    });

  });


  // Initial fetch

  fetchVideos();
